    <script>
        const SUPABASE_URL = "https://ibsztkfrkumibvllxhqn.supabase.co"; 
        const SUPABASE_KEY = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6Imlic3p0a2Zya3VtaWJ2bGx4aHFuIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzE4NzYxMTYsImV4cCI6MjA4NzQ1MjExNn0.igRwNr_mNV4NVWlo_DzpFZ3jbnMPvqmhlX7FRbs4bKs"; 
        const supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
        
        const TABLE_NAME = 'vacaciones_demo'; 
        const MAX_RESOURCES_PER_DAY = 2;

        let calendar, selectedRange, currentUser, vacacionesData = [];
        let currentYear = 2026; 
        let isLocalMode = false;

        const employeeConfig = {
            "User Test 1": { color: "#10b981", grupo: "admin" },
            "User Test 2": { color: "#3b82f6", grupo: "admin" },
            "Mecánico A": { color: "#f97316", grupo: "taller" },
            "Mecánico B": { color: "#06b6d4", grupo: "taller" }

        const employees = Object.keys(employeeConfig);

        function showToast(msg) {
            const t = document.getElementById('toast');
            document.getElementById('toastMsg').innerText = msg;
            t.classList.add('show');
            setTimeout(() => t.classList.remove('show'), 4000);
        }

        function handleLogin() {
            const userIn = document.getElementById('user').value.trim().toLowerCase();
            const passIn = document.getElementById('pass').value.trim();
            if(userIn === 'admin' && passIn === 'metal2026') {
                document.getElementById('loginScreen').style.display = 'none';
                document.getElementById('mainApp').style.display = 'grid';
                initApp();
            } else {
                showToast("Credenciales de Demo incorrectas");
            }
        }

        async function initApp() {
            initCalendar();
            await fetchData();
        }

        async function fetchData() {
            const { data, error } = await supabaseClient.from(TABLE_NAME).select('*');
            
            if(error && (error.code === "PGRST205" || error.message.includes("not found"))) {
                isLocalMode = true;
                document.getElementById('localModeWarning').style.display = 'block';
                const localData = localStorage.getItem('vacaciones_demo_local');
                vacacionesData = localData ? JSON.parse(localData) : [];
            } else if (error) {
                console.error("Error de Supabase:", error);
                showToast("Error de conexión");
            } else {
                isLocalMode = false;
                document.getElementById('localModeWarning').style.display = 'none';
                vacacionesData = data || [];
            }

            updateCalendarEvents();
            renderUserList();
            if(currentUser) renderTable();
        }

        function initCalendar() {
            calendar = new FullCalendar.Calendar(document.getElementById('calendar'), {
                initialView: 'multiMonthYear',
                initialDate: `${currentYear}-01-01`,
                locale: 'es',
                firstDay: 1,
                selectable: true,
                headerToolbar: false,
                select: (info) => {
                    if(!currentUser) {
                        showToast("Selecciona un empleado primero");
                        calendar.unselect();
                        return;
                    }

                    selectedRange = info;
                    const startSelected = new Date(info.startStr);
                    
                    // LÓGICA DE TOGGLE: ¿Hay un registro del usuario actual que coincida con este día?
                    // Buscamos cualquier registro del usuario activo que cubra al menos una de las fechas seleccionadas
                    const overlappingRecord = vacacionesData.find(v => {
                        if (v.empleado !== currentUser) return false;
                        
                        const recordStart = new Date(v.fecha_inicio);
                        const recordEnd = new Date(v.fecha_fin);
                        const selStart = new Date(info.start);
                        const selEnd = new Date(info.end);
                        selEnd.setDate(selEnd.getDate() - 1); // Ajuste fin de FullCalendar

                        // Si hay solapamiento entre el rango del registro y el rango seleccionado
                        return (selStart <= recordEnd && selEnd >= recordStart);
                    });

                    if (overlappingRecord) {
                        // Si ya existe registro, mostramos el modal de borrado en lugar del de añadir
                        document.getElementById('confirmDeleteBtn').onclick = () => deleteRecord(overlappingRecord.id);
                        document.getElementById('deleteModal').style.display = 'flex';
                    } else {
                        // Si no existe, procedemos a añadir
                        const fFinShow = new Date(info.end);
                        fFinShow.setDate(fFinShow.getDate() - 1);
                        const fFinStr = fFinShow.toISOString().split('T')[0];
                        
                        document.getElementById('selectedDatesLabel').innerText = `${info.startStr} hasta ${fFinStr}`;
                        document.getElementById('modal').style.display = 'flex';
                    }
                }
            });
            calendar.render();
        }

        function updateCalendarEvents() {
            calendar.removeAllEvents();
            const events = vacacionesData.map(v => {
                const endDate = new Date(v.fecha_fin);
                endDate.setDate(endDate.getDate() + 1);
                
                return {
                    id: v.id.toString(),
                    title: v.empleado.split(' ')[0],
                    start: v.fecha_inicio, 
                    end: endDate.toISOString().split('T')[0],
                    backgroundColor: employeeConfig[v.empleado]?.color || '#94a3b8',
                    allDay: true
                };
            });
            calendar.addEventSource(events);
        }

        async function saveSelection(factor) {
            const fInicioStr = selectedRange.startStr.split('T')[0];
            const fFinDate = new Date(selectedRange.end);
            fFinDate.setDate(fFinDate.getDate() - 1);
            const fFinStr = fFinDate.toISOString().split('T')[0];

            // VALIDACIÓN DE CUPO MÁXIMO
            let currCheck = new Date(selectedRange.start);
            const endCheck = new Date(selectedRange.end);
            let dayViolated = null;

            while(currCheck < endCheck) {
                const dateToCheck = currCheck.toISOString().split('T')[0];
                const count = vacacionesData.filter(v => {
                    const dStart = new Date(v.fecha_inicio);
                    const dEnd = new Date(v.fecha_fin);
                    const dCheck = new Date(dateToCheck);
                    return dCheck >= dStart && dCheck <= dEnd;
                }).length;

                if (count >= MAX_RESOURCES_PER_DAY) {
                    dayViolated = dateToCheck;
                    break;
                }
                currCheck.setDate(currCheck.getDate() + 1);
            }

            if (dayViolated) {
                showToast(`⚠️ CUPO LLENO: El día ${dayViolated} ya tiene ${MAX_RESOURCES_PER_DAY} recursos.`);
                closeModal();
                return;
            }

            let days = 0; 
            let curr = new Date(selectedRange.start);
            const end = new Date(selectedRange.end);
            while(curr < end) {
                if(curr.getDay() !== 0 && curr.getDay() !== 6) days += factor;
                curr.setDate(curr.getDate() + 1);
            }

            if(days <= 0) {
                showToast("Días no laborables");
                closeModal();
                return;
            }

            const record = { empleado: currentUser, fecha_inicio: fInicioStr, fecha_fin: fFinStr, dias: days };

            if(isLocalMode) {
                record.id = Date.now().toString();
                vacacionesData.push(record);
                localStorage.setItem('vacaciones_demo_local', JSON.stringify(vacacionesData));
            } else {
                const { error } = await supabaseClient.from(TABLE_NAME).insert([record]);
                if(error) { showToast("Error al grabar"); return; }
            }

            await fetchData(); 
            closeModal();
            showToast("Registro guardado");
        }

        function renderUserList() {
            ['list-admin', 'list-taller'].forEach(id => document.getElementById(id).innerHTML = '');
            employees.forEach(name => {
                const used = vacacionesData.filter(v => v.empleado === name).reduce((s,v)=>s+v.dias,0);
                const html = `
                    <div onclick="selectUser('${name}')" class="p-3 rounded-xl cursor-pointer transition-all ${currentUser === name ? 'active-user' : 'hover:bg-slate-50'}">
                        <div class="flex justify-between items-center text-xs font-bold">
                            <span>${name}</span><span>${used.toFixed(1)}/25</span>
                        </div>
                        <div class="progress-bar-bg"><div class="progress-bar-fill" style="width:${(used/25)*100}%; background:${employeeConfig[name].color}"></div></div>
                    </div>`;
                document.getElementById(`list-${employeeConfig[name].grupo}`).innerHTML += html;
            });
        }

        function selectUser(name) {
            currentUser = name;
            document.getElementById('currentUserTitle').innerText = name;
            document.getElementById('registrosContainer').classList.remove('hidden');
            renderUserList(); 
            renderTable();
        }

        function renderTable() {
            const filtered = vacacionesData.filter(v => v.empleado === currentUser);
            document.getElementById('tableBody').innerHTML = filtered.map(v => `
                <tr class="border-b hover:bg-slate-50 transition-colors">
                    <td class="p-4 text-xs font-bold text-slate-600 uppercase tracking-tight">${v.fecha_inicio} <span class="text-slate-300 mx-2">→</span> ${v.fecha_fin}</td>
                    <td class="p-4 text-center font-black text-red-600">${v.dias}d</td>
                    <td class="p-4 text-right">
                        <button onclick="deleteRecord('${v.id}')" class="bg-red-50 text-red-500 px-3 py-1 rounded-lg text-[10px] font-black uppercase hover:bg-red-500 hover:text-white transition-all">Borrar</button>
                    </td>
                </tr>`).join('');
        }

        async function deleteRecord(id) {
            if(isLocalMode) {
                vacacionesData = vacacionesData.filter(v => v.id.toString() !== id.toString());
                localStorage.setItem('vacaciones_demo_local', JSON.stringify(vacacionesData));
            } else {
                const { error } = await supabaseClient.from(TABLE_NAME).delete().eq('id', id);
                if(error) { showToast("Error al borrar"); return; }
            }
            await fetchData();
            closeModal();
            showToast("Registro eliminado");
        }

        async function clearDemoTable() {
            if(confirm("¿Borrar todos los datos de prueba?")) {
                if(isLocalMode) { localStorage.removeItem('vacaciones_demo_local'); }
                else { await supabaseClient.from(TABLE_NAME).delete().neq('empleado', 'SISTEMA_RESERVADO'); }
                vacacionesData = [];
                await fetchData();
                showToast("Datos reseteados");
            }
        }

        function navYear(offset) {
            currentYear += offset;
            document.getElementById('displayYear').innerText = currentYear;
            calendar.gotoDate(`${currentYear}-01-01`);
            fetchData();
        }

        function closeModal() { 
            document.getElementById('modal').style.display = 'none'; 
            document.getElementById('deleteModal').style.display = 'none'; 
            calendar.unselect(); 
        }
