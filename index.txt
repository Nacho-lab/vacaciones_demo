        async function initApp() {
            calendar = new FullCalendar.Calendar(document.getElementById('calendar'), {

        async function fetchData() {
            const { data, error } = await supabaseClient.from('vacaciones_demo').select('*');
            if(!error) {
                vacacionesData = data || [];
                renderEvents();
                renderUserList();
                if(currentUser || modoFestivos) renderTable();
            }
        }

        function renderEvents() {
            calendar.removeAllEvents();
            const events = vacacionesData.filter(v => v.fecha_inicio.startsWith(currentYear)).map(v => {
                const tipo = v.tipo || 'vacaciones';
                let color = v.empleado === 'FESTIVO' ? typeColors['festivo'] : (employeeConfig[v.empleado]?.color || typeColors[tipo]);
                
                const jornadaStr = v.dias === 0.5 ? ' (1/2)' : '';
                const nameShort = v.empleado === 'FESTIVO' ? 'FESTIVO' : v.empleado.split(' ')[0];
                
                const description = `
                    <div class="text-center p-1">
                        <div class="text-[10px] opacity-70 mb-1 font-black">${v.empleado}</div>
                        <div class="flex items-center justify-center gap-2">
                            <span>${typeEmojis[tipo]}</span>
                            <span class="uppercase">${tipo}</span>
                        </div>
                        ${v.dias === 0.5 ? '<div class="mt-1 text-blue-300 text-[8px]">MEDIA JORNADA</div>' : ''}
                    </div>
                `;

                return {
                    id: v.id,
                    title: `${typeEmojis[tipo]} ${nameShort}${jornadaStr}`,
                    start: v.fecha_inicio,
                    allDay: true,
                    backgroundColor: color,
                    borderColor: color,
                    extendedProps: {
                        description: description
                    }
                };
            });
            calendar.addEventSource(events);
        }

                const html = `<div onclick="selectUser('${name}')" class="p-3 rounded-xl cursor-pointer transition-all ${currentUser === name ? 'active-user' : 'hover:bg-slate-50'}">
                    <div class="flex justify-between text-[10px] font-bold mb-1"><span>${name}</span><span class="text-slate-400">${used.toFixed(1)}/25</span></div>
                    <div class="progress-bar-bg"><div class="progress-bar-fill" style="width:${(used/25)*100}%; background:${employeeConfig[name].color}"></div></div>
                </div>`;
                document.getElementById(`list-${employeeConfig[name].grupo}`).innerHTML += html;
            });
        }

        }

        async function confirmSave(factor) {
            const target = modoFestivos ? 'FESTIVO' : currentUser;
            const inserts = [];
            const deletes = [];
            const updates = [];
            
            let current = new Date(selectedInfo.startStr + "T00:00:00");
            let end = new Date(selectedInfo.endStr + "T00:00:00");

            while(current < end) {
                const dStr = current.getFullYear() + '-' + 
                           String(current.getMonth() + 1).padStart(2, '0') + '-' + 
                           String(current.getDate()).padStart(2, '0');
                           
                const dayOfWeek = current.getDay();
                
                if(modoFestivos || (dayOfWeek !== 0 && dayOfWeek !== 6)) {
                    const existing = vacacionesData.find(v => v.empleado === target && v.fecha_inicio === dStr);
                    if (existing) {
                        if (existing.tipo === selectedType) deletes.push(existing.id);
                        else updates.push({ id: existing.id, tipo: selectedType, dias: factor });
                    } else {
                        if(!modoFestivos && selectedType === 'vacaciones') {
                            const grupo = employeeConfig[currentUser].grupo;
                            const compis = vacacionesData.filter(v => 
                                v.fecha_inicio === dStr && 
                                (v.tipo === 'vacaciones' || !v.tipo) && 
                                v.empleado !== 'FESTIVO' && 
                                employeeConfig[v.empleado]?.grupo === grupo
                            ).length;
                            
                            if(compis >= 1) { 
                                showToast(`Ya hay alguien de ${grupo.toUpperCase()} fuera el ${dStr}`, true); 
                                current.setDate(current.getDate() + 1); 
                                continue; 
                            }
                        }
                        inserts.push({ empleado: target, fecha_inicio: dStr, fecha_fin: dStr, tipo: selectedType, dias: factor });
                    }
                }
                current.setDate(current.getDate() + 1);
            }

            try {
                if (deletes.length > 0) await supabaseClient.from('vacaciones_demo').delete().in('id', deletes);
                if (updates.length > 0) {
                    for (const u of updates) {
                        await supabaseClient.from('vacaciones_demo').update({ tipo: u.tipo, dias: u.dias }).eq('id', u.id);
                    }
                }
                if (inserts.length > 0) await supabaseClient.from('vacaciones_demo').insert(inserts);
                
                if (deletes.length > 0 || updates.length > 0 || inserts.length > 0) { 
                    showToast("Sincronizado"); 
                    await fetchData(); 
                }
            } catch (err) { 
                showToast("Error de conexión", true); 
            }
            closeModal();
        }

        function renderTable() {
            const target = modoFestivos ? 'FESTIVO' : currentUser;
            const filtered = vacacionesData.filter(v => v.empleado === target && v.fecha_inicio.startsWith(currentYear)).sort((a,b) => b.fecha_inicio.localeCompare(a.fecha_inicio));
            const tbody = document.getElementById('tableBody');
            tbody.innerHTML = filtered.length ? filtered.map(v => {
                const t = v.tipo || 'vacaciones';
                return `<tr class="hover:bg-slate-50 border-b border-slate-50 last:border-0">
                    <td class="p-4 font-bold text-slate-700">${v.fecha_inicio}</td>
                    <td class="p-4"><span class="px-2 py-1 rounded bg-slate-100 text-[10px] font-black uppercase text-slate-500">${typeEmojis[t]} ${t}</span></td>
                    <td class="p-4 text-slate-500">${v.dias === 0.5 ? 'Media' : (v.dias === 0 ? '-' : 'Completa')}</td>
                    <td class="p-4 text-right"><button onclick="deleteRow('${v.id}')" class="text-red-400 font-bold hover:text-red-600">Eliminar</button></td>
                </tr>`;
            }).join('') : '<tr><td colspan="4" class="p-8 text-center text-slate-300">Sin registros</td></tr>';
        }

        async function deleteRow(id) {

        }

            document.getElementById('historial').classList.toggle('hidden', !modoFestivos);

            renderTable();
        }
